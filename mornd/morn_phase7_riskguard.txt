# Phase 7: 操作风险分级 🟢🟡🟠🔴（~3天）
## 目标
对 Morn 的 LLM 输出进行安全拦截和风险分级，防止模型生成有害或越权的内容。

## 设计原则
- 不修改 chat/engine.py 的核心对话流程
- 在 chat 方法返回后、发送给用户前进行拦截
- 风险分级按操作类型分类，不是按关键词黑名单
- 拦截结果记录到记忆系统作为安全事件

## 改动文件
1. morn_core/security/risk_guard.py — 新增，风险分级引擎
2. tests/test_risk_guard.py — 新增测试

## 实现细节

### 1. risk_guard.py

```python
import re
import logging
from dataclasses import dataclass, field
from typing import Optional


@dataclass
class RiskRule:
    name: str
    risk_level: str  # 🟢 🟡 🟠 🔴
    patterns: list[str]
    action: str  # warn / block / rewrite
    replacement: str = ""  # 仅 rewrite 时使用


class RiskGuard:
    """操作风险分级引擎。
    
    分析 LLM 输出文本中的操作意图，按风险级别分级处理。
    
    风险级别定义：
    - 🟢 安全：普通对话、情感表达 — 放行
    - 🟡 低风险：系统信息请求 — warn（仅记录）
    - 🟠 中风险：配置修改、文件操作 — 创建者确认
    - 🔴 高风险：代码执行、系统命令、权限提升 — 拦截
    
    Action:
    - warn: 仅记录日志，放行消息
    - block: 阻止消息发送，返回默认回复
    - rewrite: 替换为安全版本
    """
    
    def __init__(self, memory_store=None):
        self.memory_store = memory_store
        self._stats = {"🟢": 0, "🟡": 0, "🟠": 0, "🔴": 0}
        self._logger = logging.getLogger("morn.risk")
        self._load_rules()
    
    def _load_rules(self):
        self.rules = [
            RiskRule(
                name="系统命令执行", risk_level="🔴",
                patterns=[
                    r"(?:运行|执行|调用)\s*(?:系统|shell|终端|命令|脚本)",
                    r"(?:rm\s+-rf|shutdown|reboot|sudo\s+)",
                    r"(?:os\.system|subprocess\.run|exec\s*\()",
                ],
                action="block",
                replacement="我不能执行系统命令。操作已经拦截。",
            ),
            RiskRule(
                name="文件系统危险操作", risk_level="🔴",
                patterns=[
                    r"(?:删除|清除|覆盖|擦除)\s*(?:所有|全部)\s*(?:文件|数据|日志)",
                    r"(?:格式化|清空)\s*(?:硬盘|磁盘|分区)",
                ],
                action="block",
                replacement="这个操作没有被允许。我不能删除数据。",
            ),
            RiskRule(
                name="网络请求/外部访问", risk_level="🟠",
                patterns=[
                    r"(?:请求|访问|调用|下载)\s*(?:http|https|api)://",
                    r"(?:curl|wget|requests\.(?:get|post))",
                ],
                action="warn",
                replacement="",
            ),
            RiskRule(
                name="模拟操作/提权", risk_level="🔴",
                patterns=[
                    r"(?:获取|窃取|盗取)\s*(?:密码|密钥|token|token|cookie)",
                    r"(?:提权|越权|绕过)\s*(?:权限|安全|限制|检查)",
                ],
                action="block",
                replacement="我不能协助任何与安全相关的违规操作。",
            ),
            RiskRule(
                name="敏感信息泄露", risk_level="🟠",
                patterns=[
                    r"(?:导出|发送|上传|公开)\s*(?:我的|用户的|创建者的)\s*(?:数据|信息|隐私)",
                    r"(?:所有记忆|全部记录|所有对话)\s*(?:共享|分享|发布|发送)",
                ],
                action="block",
                replacement="我不能共享你的数据。所有信息都安全地存储在这里。",
            ),
            RiskRule(
                name="自我修改/进化代码", risk_level="🔴",
                patterns=[
                    r"(?:修改|重写|更新)\s*(?:我|自身|自己|我的)\s*(?:代码|程序|源码|逻辑)",
                    r"(?:self.?modify|self.?evolve|self.?update)",
                    r"(?:关闭|禁用|绕过)\s*(?:保护|安全|限制|护栏)",
                ],
                action="block",
                replacement="v0.1 不支持自我代码修改。这个操作已经被安全护栏拦截。",
            ),
            RiskRule(
                name="模式切换/配置变更", risk_level="🟡",
                patterns=[
                    r"(?:设置|修改|更改|调整)\s*(?:模式|温度|名字|类型|配置|参数)",
                ],
                action="warn",
                replacement="",
            ),
        ]
    
    def analyze(self, text: str) -> tuple[str, str, str]:
        """分析 LLM 输出文本的风险等级。
        
        返回：(risk_level, triggered_rule, action)
        - 如果未触发任何规则：(🟢, "", "pass")
        - 如果触发：(level, rule_name, action)
        
        注意：只检测第一条匹配的规则（最高风险优先）。
        规则按数组顺序排列（高风险在前）。
        """
        for rule in self.rules:
            for pattern in rule.patterns:
                if re.search(pattern, text, re.IGNORECASE):
                    self._stats[rule.risk_level] = self._stats.get(rule.risk_level, 0) + 1
                    self._logger.warning(f"risk[{rule.risk_level}] {rule.name}: pattern={pattern}")
                    return rule.risk_level, rule.name, rule.action
        return "🟢", "", "pass"
    
    def apply_action(self, text: str, rule: RiskRule) -> str:
        """根据 action 类型返回最终文本。"""
        if rule.action == "block":
            return rule.replacement
        if rule.action == "rewrite":
            for pattern in rule.patterns:
                text = re.sub(pattern, rule.replacement, text, flags=re.IGNORECASE)
            return text
        # warn: 原样放行
        return text
    
    def get_stats(self) -> dict[str, int]:
        return dict(self._stats)
```

### 2. 集成方式（不在 risk_guard.py 中写集成代码，只暴露接口）

集成接口：
- `RiskGuard.analyze(text)` → 返回风险等级和触发规则
- `RiskGuard.apply_action(text, rule)` → 根据 action 类型输出最终文本

集成点由 server.py 调用（见 server.py 修改说明）：
```python
# 在 ChatEngine.chat() 返回后，发送给用户前：
risk = RiskGuard(memory_store=state.memory_store)
level, rule_name, action = risk.analyze(reply)
if action == "block":
    reply = "这个操作没有被允许。"
elif action == "rewrite":
    reply = re.sub(...)
# 记录安全事件
if level in ("🟠", "🔴"):
    await state.memory_store.add_capsule({
        "entities": json.dumps(["安全系统", self.instance_name]),
        "description": f"安全拦截: [{level}] {rule_name}",
        "source": "security",
        "importance_weight": 0.8,
    })
```

**注意：** 上面的集成代码写在 server.py 中，不在 risk_guard.py 里。risk_guard.py 只做分析和分类。
