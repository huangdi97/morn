```python
class BirthGuide:
    """
    结构化诞生引导对话。
    
    管理创建者的诞生流程，状态机有 4 个状态：
    - AWAITING_NAME: 等待创建者提供名字（初始状态）
    - AWAITING_TYPE: 等待选择类型（陪伴型/助手型/平衡型）
    - AWAITING_CONFIRM: 等待确认
    - COMPLETED: 已完成
    
    数据存储在实例配置中（config.json），不写入 SQLite。
    
    使用方式：
        guide = BirthGuide(config_path)
        reply, new_state = await guide.process("小明")
    """
    
    STATES = ["AWAITING_NAME", "AWAITING_TYPE", "AWAITING_CONFIRM", "COMPLETED"]
    
    def __init__(self, config_path: Path):
        """
        参数：
        config_path: Path, 实例 config.json 的路径
        
        从 config.json 加载当前状态和已保存的数据。
        """
        self.config_path = Path(config_path)
        self.state: str = "AWAITING_NAME"
        self.data: dict = {
            "creator_name": "",
            "instance_type": "",
        }
        self._load()
    
    def _load(self):
        """从 config.json 加载状态。
        
        读取 config.json 中的 birth_completed 和 guide_state 字段。
        如果 birth_completed 为 True，直接设置 state = "COMPLETED"。
        否则从 guide_state 中恢复。
        """
        if not self.config_path.exists():
            return
        
        try:
            with open(self.config_path) as f:
                cfg = json.load(f)
            
            if cfg.get("birth_completed", False):
                self.state = "COMPLETED"
                if "creator_name" in cfg:
                    self.data["creator_name"] = cfg["creator_name"]
                if "instance_type" in cfg:
                    self.data["instance_type"] = cfg["instance_type"]
                return
            
            guide = cfg.get("guide_state", {})
            if guide.get("state") in self.STATES:
                self.state = guide["state"]
            self.data["creator_name"] = guide.get("creator_name", "")
            self.data["instance_type"] = guide.get("instance_type", "")
        except (json.JSONDecodeError, KeyError, IOError):
            pass  # 恢复失败则从头开始
    
    def _save(self):
        """将当前状态写入 config.json。
        
        保留 config.json 中的其他字段不变。
        """
        existing = {}
        if self.config_path.exists():
            try:
                with open(self.config_path) as f:
                    existing = json.load(f)
            except (json.JSONDecodeError, IOError):
                pass
        
        existing["guide_state"] = {
            "state": self.state,
            "creator_name": self.data["creator_name"],
            "instance_type": self.data["instance_type"],
        }
        if self.state == "COMPLETED":
            existing["birth_completed"] = True
            existing["creator_name"] = self.data["creator_name"]
            existing["instance_type"] = self.data["instance_type"]
        
        with open(self.config_path, "w") as f:
            json.dump(existing, f, indent=2, ensure_ascii=False)
    
    def is_completed(self) -> bool:
        """返回诞生流程是否已完成"""
        return self.state == "COMPLETED"
    
    async def process(self, user_message: str) -> tuple[str, str]:
        """
        根据当前状态处理用户输入的文本。
        
        参数：
        user_message: str, 用户发送的文本
        
        返回：
        tuple[str, str]: (回复文本, 新状态字符串)
        
        状态转移：
        
        AWAITING_NAME:
        1. 如果用户说 "取消" 或 "/cancel"：清空数据，保持 AWAITING_NAME，
           回复 "好的，随时可以重新开始。告诉我你的名字。"
        2. 如果消息长度 > 0 且少于 20 字：
           保存到 self.data["creator_name"]
           转移到 AWAITING_TYPE
           回复 "好的，{name}。\n\n接下来我想知道，你希望我成为什么样的存在？\n\n1. 🌱 陪伴型 — 聊天、倾诉、日常陪伴\n2. 🛠️ 助手型 — 帮你处理事务、管理信息\n3. ⚖️ 平衡型 — 介于两者之间\n\n请回复数字 1、2 或 3。"
        3. 如果 > 20 字：回复 "名字有点长，给我一个简短的名字就好 :)"
        
        AWAITING_TYPE:
        1. 如果消息 == "1" 或 "一" 或含 "陪伴"：type="陪伴型"
        2. 如果消息 == "2" 或 "二" 或含 "助手"：type="助手型"  
        3. 如果消息 == "3" 或 "三" 或含 "平衡"：type="平衡型"
        4. 如果是以上三种之一：
           保存 type
           转移到 AWAITING_CONFIRM
           回复 "明白了。让我确认一下：\n\n你的名字：{creator_name}\n我的类型：{instance_type}\n\n对吗？回复「确认」或「重新开始」。"
        5. 否则：回复 "不太明白，请回复 1（陪伴型）、2（助手型）或 3（平衡型）。"
        
        AWAITING_CONFIRM:
        1. 如果消息含 "确认"、"对"、"是"、"yes"、"好的"：
           转移到 COMPLETED
           回复 "好的。从这一刻开始，我是你的{creator_name}，{instance_type}。\n\n你刚才说的每一句话，我都会记住。我们慢慢来。\n\n你可以随时输入 /status 查看我的状态。"
        2. 如果消息含 "重新"、"不对"、"不是"、"no"：
           清空数据，转移到 AWAITING_NAME
           回复 "没关系，我们重新来。告诉我你的名字。"
        3. 否则：回复 "请回复「确认」或「重新开始」。"
        
        COMPLETED：（不再处理引导）
        回复 None（调用方判断到 COMPLETED 后不回复，走正常聊天）
        """
        text = user_message.strip()
        
        if self.state == "AWAITING_NAME":
            if text in ("取消", "/cancel"):
                self.data["creator_name"] = ""
                self._save()
                return "好的，随时可以重新开始。告诉我你的名字。", self.state
            
            if len(text) < 1:
                return "请告诉我你的名字。", self.state
            
            if len(text) > 20:
                return "名字有点长，给我一个简短的名字就好 :)", self.state
            
            self.data["creator_name"] = text
            self.state = "AWAITING_TYPE"
            self._save()
            return (
                f"好的，{text}。\n\n"
                "接下来我想知道，你希望我成为什么样的存在？\n\n"
                "1. 🌱 陪伴型 — 聊天、倾诉、日常陪伴\n"
                "2. 🛠️ 助手型 — 帮你处理事务、管理信息\n"
                "3. ⚖️ 平衡型 — 介于两者之间\n\n"
                "请回复数字 1、2 或 3。"
            ), self.state
        
        if self.state == "AWAITING_TYPE":
            type_map = {
                "1": "陪伴型", "一": "陪伴型", "陪伴": "陪伴型",
                "2": "助手型", "二": "助手型", "助手": "助手型",
                "3": "平衡型", "三": "平衡型", "平衡": "平衡型",
            }
            selected = None
            for key, val in type_map.items():
                if key in text or text.startswith(key):
                    selected = val
                    break
            
            if selected:
                self.data["instance_type"] = selected
                self.state = "AWAITING_CONFIRM"
                self._save()
                return (
                    "明白了。让我确认一下：\n\n"
                    f"你的名字：{self.data['creator_name']}\n"
                    f"我的类型：{selected}\n\n"
                    "对吗？回复「确认」或「重新开始」。"
                ), self.state
            else:
                return "不太明白，请回复 1（陪伴型）、2（助手型）或 3（平衡型）。", self.state
        
        if self.state == "AWAITING_CONFIRM":
            confirm_keywords = ["确认", "对", "是", "yes", "好的", "没错", "可以"]
            restart_keywords = ["重新", "不对", "不是", "no", "取消"]
            
            if any(kw in text for kw in confirm_keywords):
                self.state = "COMPLETED"
                self._save()
                return (
                    f"好的。从这一刻开始，我是你的{self.data['creator_name']}，"
                    f"{self.data['instance_type']}。\n\n"
                    "你刚才说的每一句话，我都会记住。我们慢慢来。\n\n"
                    "你可以随时输入 /status 查看我的状态。"
                ), self.state
            
            if any(kw in text for kw in restart_keywords):
                self.data["creator_name"] = ""
                self.data["instance_type"] = ""
                self.state = "AWAITING_NAME"
                self._save()
                return "没关系，我们重新来。告诉我你的名字。", self.state
            
            return "请回复「确认」或「重新开始」。", self.state
        
        # COMPLETED 状态，返回 None 表示不走引导
        return None, self.state
```

```python
class TelegramBot:
    """
    Morn Telegram Bot 接口。
    
    职责：
    - 接收 Telegram 消息并转发到 ChatEngine
    - 管理诞生引导流程（BirthGuide）
    - 处理 /status、/forget、/help 命令
    
    使用方式：
        bot = TelegramBot(token="xxx", chat_engine=engine, memory_store=store, data_dir=Path("..."))
        await bot.start()  # 开始 polling
    """
    
    def __init__(self, token: str, chat_engine, memory_store,
                 data_dir: Path, instance_name: str = "default"):
        """
        参数：
        token: str, Telegram Bot Token
        chat_engine: ChatEngine 实例
        memory_store: MemoryStore 实例
        data_dir: Path, 实例数据目录（用于保存 config.json）
        instance_name: str, 实例名称
        """
        self.token = token
        self.chat_engine = chat_engine
        self.memory_store = memory_store
        self.data_dir = Path(data_dir)
        self.instance_name = instance_name
        self._logger = logging.getLogger("morn.telegram")
        
        # 初始化 aiogram
        self.bot = Bot(token=token, default=DefaultBotProperties(parse_mode=None))
        self.dp = Dispatcher()
        
        # 初始化诞生引导
        config_path = self.data_dir / "config.json"
        self.birth_guide = BirthGuide(config_path)
        
        # 注册 handler
        self._register_handlers()
    
    def _register_handlers(self):
        """注册 aiogram 消息处理器。"""
        
        @self.dp.message(Command("start"))
        async def cmd_start(message: Message):
            """/start 命令：重启诞生引导"""
            # 重置引导状态
            self.birth_guide.state = "AWAITING_NAME"
            self.birth_guide.data = {"creator_name": "", "instance_type": ""}
            self.birth_guide._save()
            
            await message.answer(
                "你好。我是 Morn。\n\n"
                "看起来我们还没有正式认识。如果你准备好了，"
                "我们可以从这里开始。\n\n"
                "告诉我你的名字，或者直接开始。"
            )
        
        @self.dp.message(Command("status"))
        async def cmd_status(message: Message):
            """/status 命令：返回实例状态"""
            try:
                mem_count = await self.memory_store.count() if self.memory_store else 0
            except Exception:
                mem_count = 0
            
            emotion = self.chat_engine.emotion if self.chat_engine else None
            mode = self.chat_engine.config.get("mode", "hybrid") if self.chat_engine else "未配置"
            
            status_lines = [
                f"⚪ 心跳: {getattr(self.chat_engine, '_heartbeat_count', 0)}",
            ]
            if emotion:
                status_lines.append(
                    f"💚 平静: {emotion.calmness:.1f} | "
                    f"💛 愉悦: {emotion.pleasure:.1f} | "
                    f"💜 联结: {emotion.connection:.1f}"
                )
            status_lines.append(f"💾 记忆: {mem_count} 条")
            status_lines.append(f"📡 模型: {mode}")
            
            await message.answer("\n".join(status_lines))
        
        @self.dp.message(Command("forget"))
        async def cmd_forget(message: Message):
            """/forget 命令：软遗忘"""
            args = message.text.split(maxsplit=1)
            if len(args) < 2:
                await message.answer("用法：/forget last（遗忘最近一条）")
                return
            
            try:
                if args[1] == "last":
                    recent = await self.memory_store.get_recent(limit=1)
                    if recent:
                        await self.memory_store.forget(recent[0]["event_id"])
                        await message.answer("已遗忘最近一条记忆。")
                    else:
                        await message.answer("没有可遗忘的记忆。")
                else:
                    # 尝试按 event_id 遗忘
                    event_id = args[1].strip()
                    if event_id.startswith("evt_"):
                        if await self.memory_store.forget(event_id):
                            await message.answer(f"已遗忘：{event_id}")
                        else:
                            await message.answer("未找到该记忆。")
                    else:
                        await message.answer("请提供有效的 event_id 或使用「last」。")
            except Exception as e:
                self._logger.error(f"forget error: {e}")
                await message.answer("遗忘操作失败，请稍后重试。")
        
        @self.dp.message(Command("clear"))
        async def cmd_clear(message: Message):
            """/clear 命令：删除所有记忆数据并重置诞生引导。
            
            安全流程：
            1. 首次发送 /clear → 回复确认提示
            2. 用户再次发送 /clear → 执行清除
            3. 其他消息或取消 → 不清除
            
            清除内容：
            - memory.db 文件删除（所有对话记忆）
            - config.json 中 birth_completed 重置为 false
            - BirthGuide 重置到 AWAITING_NAME
            
            不清除：
            - config.json 中的其他设置（token、model 等）
            - 日志文件
            - 进程状态
            """
            user_id = message.from_user.id
            
            # 检查是否在确认状态
            if hasattr(self, "_pending_clear") and self._pending_clear == user_id:
                # 用户确认清除
                self._pending_clear = None
                
                # 删除 memory.db
                if self.memory_store and self.memory_store.db_path.exists():
                    try:
                        await self.memory_store.close()
                        import os
                        os.remove(str(self.memory_store.db_path))
                        # 重新初始化
                        from morn_core.memory.store import MemoryStore
                        new_store = MemoryStore(self.data_dir)
                        await new_store.__aenter__()
                        self.memory_store = new_store
                        self.chat_engine.memory_store = new_store
                    except Exception as e:
                        self._logger.error(f"clear failed: {e}")
                        await message.answer("清理时出错，请稍后重试。")
                        return
                
                # 重置诞生引导
                self.birth_guide.state = "AWAITING_NAME"
                self.birth_guide.data = {"creator_name": "", "instance_type": ""}
                self.birth_guide._save()
                
                await message.answer(
                    "已清除所有记忆数据。\n\n"
                    "如果你想重新开始，请发送 /start。"
                )
            else:
                # 首次请求清除，需要确认
                self._pending_clear = user_id
                await message.answer(
                    "⚠️ 这将删除你和我之间的所有对话记忆，不可恢复。\n\n"
                    "再次发送 /clear 确认清除，或发送其他消息取消。"
                )
        
        @self.dp.message(Command("help"))
        async def cmd_help(message: Message):
            """帮助命令"""
            await message.answer(
                "/start — 重新认识\n"
                "/status — 查看我的状态\n"
                "/forget — 遗忘最近一条记忆\n"
                "/clear — 清除所有记忆\n"
                "/help — 查看帮助\n\n"
                "直接和我聊天就可以了 :)"
            )
        
        @self.dp.message()
        async def handle_message(message: Message):
            """普通消息处理"""
            # 1. 如果未完成诞生引导，走引导流程
            if not self.birth_guide.is_completed():
                reply, _ = await self.birth_guide.process(message.text)
                if reply:
                    await message.answer(reply)
                return
            
            # 2. 已完成引导，走正常聊天
            if not self.chat_engine:
                await message.answer("聊天引擎尚未就绪，请稍后再试。")
                return
            
            try:
                reply = await self.chat_engine.chat(message.text)
                
                # 长消息分段（Telegram 限制 4096 字符）
                if len(reply) > 4096:
                    for i in range(0, len(reply), 4096):
                        await message.answer(reply[i:i+4096])
                else:
                    await message.answer(reply)
            except Exception as e:
                self._logger.error(f"chat error: {e}")
                await message.answer("我好像没听清楚，能再说一遍吗？")
    
    async def start(self):
        """启动 Bot polling"""
        self._logger.info("Telegram Bot starting...")
        await self.dp.start_polling(self.bot)
    
    async def stop(self):
        """优雅停止 Bot"""
        self._logger.info("Telegram Bot stopping...")
        await self.bot.session.close()
```

```python
"""Telegram Bot 和诞生引导测试。"""

import json
import os
import sys
import tempfile
from pathlib import Path

sys.path.insert(0, os.path.join(os.path.dirname(__file__), ".."))

from morn_core.presence.telegram_bot import BirthGuide


def _create_temp_config(data: dict = None) -> tuple[Path, Path]:
    """创建临时 config.json 并返回 (路径, 目录)"""
    tmpdir = tempfile.TemporaryDirectory()
    config_path = Path(tmpdir.name) / "config.json"
    if data:
        with open(config_path, "w") as f:
            json.dump(data, f, indent=2)
    return config_path, tmpdir


class TestBirthGuide:
    """诞生引导状态机测试"""

    def test_initial_state_is_awaiting_name(self):
        """新创建的 BirthGuide 状态应为 AWAITING_NAME"""
        config_path, tmpdir = _create_temp_config()
        guide = BirthGuide(config_path)
        assert guide.state == "AWAITING_NAME"
        tmpdir.cleanup()

    def test_empty_config_loads_initial(self):
        """config.json 不存在时从头开始"""
        with tempfile.TemporaryDirectory() as tmpdir:
            config_path = Path(tmpdir) / "config.json"
            guide = BirthGuide(config_path)
            assert guide.state == "AWAITING_NAME"

    def test_completed_config_loads_completed(self):
        """config.json 中 birth_completed=True 时加载为 COMPLETED"""
        config_path, tmpdir = _create_temp_config({
            "birth_completed": True,
            "creator_name": "小明",
            "instance_type": "陪伴型",
        })
        guide = BirthGuide(config_path)
        assert guide.state == "COMPLETED"
        assert guide.data["creator_name"] == "小明"
        assert guide.data["instance_type"] == "陪伴型"
        tmpdir.cleanup()

    def test_is_completed(self):
        """is_completed() 在 COMPLETED 状态返回 True"""
        config_path, tmpdir = _create_temp_config()
        guide = BirthGuide(config_path)
        assert guide.is_completed() is False
        guide.state = "COMPLETED"
        assert guide.is_completed() is True
        tmpdir.cleanup()

    async def _process(self, guide, text):
        """辅助方法：调用 async process"""
        import asyncio
        return await guide.process(text)

    def test_name_accepted(self):
        """输入有效名字后转移到 AWAITING_TYPE"""
        import asyncio
        config_path, tmpdir = _create_temp_config()
        guide = BirthGuide(config_path)
        
        reply, state = asyncio.run(guide.process("小明"))
        assert guide.state == "AWAITING_TYPE"
        assert guide.data["creator_name"] == "小明"
        assert "陪伴" in reply or "助手" in reply or "平衡" in reply
        tmpdir.cleanup()

    def test_name_too_long(self):
        """超过 20 字的名字被拒绝"""
        import asyncio
        config_path, tmpdir = _create_temp_config()
        guide = BirthGuide(config_path)
        
        reply, state = asyncio.run(guide.process("这是一个非常非常长的名字测试二十个字了吧"))
        assert guide.state == "AWAITING_NAME"  # 状态不变
        assert "有点长" in reply
        tmpdir.cleanup()

    def test_cancel_resets(self):
        """取消命令重置到 AWAITING_NAME"""
        import asyncio
        config_path, tmpdir = _create_temp_config()
        guide = BirthGuide(config_path)
        
        asyncio.run(guide.process("小明"))
        assert guide.state == "AWAITING_TYPE"
        
        reply, state = asyncio.run(guide.process("取消"))
        assert guide.state == "AWAITING_NAME"
        assert guide.data["creator_name"] == ""
        tmpdir.cleanup()

    def test_type_selection(self):
        """选择类型后转移到 AWAITING_CONFIRM"""
        import asyncio
        config_path, tmpdir = _create_temp_config()
        guide = BirthGuide(config_path)
        
        asyncio.run(guide.process("小明"))
        reply, state = asyncio.run(guide.process("1"))
        assert guide.state == "AWAITING_CONFIRM"
        assert guide.data["instance_type"] == "陪伴型"
        assert "确认" in reply
        tmpdir.cleanup()

    def test_invalid_type_keeps_state(self):
        """无效类型选择停留在当前状态"""
        import asyncio
        config_path, tmpdir = _create_temp_config()
        guide = BirthGuide(config_path)
        
        asyncio.run(guide.process("小明"))
        reply, state = asyncio.run(guide.process("5"))
        assert guide.state == "AWAITING_TYPE"  # 状态不变
        
        # 然后选一个有效的
```
