**3. 新增函数：load_config()**

```python
def load_config(data_dir: Path) -> dict:
    """
    从实例目录加载或创建默认配置文件。
    
    配置文件路径：{data_dir}/config.json
    
    默认配置：
    {
        "instance_name": "default",
        "instance_type": "平衡型",
        "mode": "hybrid",
        "model_provider": "deepseek",
        "model_name": "deepseek-chat",
        "ollama_model": "qwen2.5:1.5b",
        "ollama_host": "http://localhost:11434",
        "telegram_token": "",
        "memory_retention_days": 30,
        "temperature": 0.7,
        "birth_completed": false
    }
    
    实现：
    1. 如果 config.json 不存在，用默认值创建并写入
    2. 如果存在，加载并补充缺失的默认字段
    3. 返回合并后的配置 dict
    """
    default_config = {
        "instance_name": "default",
        "instance_type": "平衡型",
        "mode": "hybrid",
        "model_provider": "deepseek",
        "model_name": "deepseek-chat",
        "ollama_model": "qwen2.5:1.5b",
        "ollama_host": "http://localhost:11434",
        "telegram_token": "",
        "memory_retention_days": 30,
        "temperature": 0.7,
        "birth_completed": False,
    }
    
    config_path = data_dir / "config.json"
    if not config_path.exists():
        data_dir.mkdir(parents=True, exist_ok=True)
        with open(config_path, "w") as f:
            json.dump(default_config, f, indent=2, ensure_ascii=False)
        return dict(default_config)
    
    try:
        with open(config_path) as f:
            cfg = json.load(f)
        # 补充缺失的默认字段
        for key, val in default_config.items():
            if key not in cfg:
                cfg[key] = val
        return cfg
    except (json.JSONDecodeError, IOError):
        return dict(default_config)
```

**4. 修改 main() 函数：串联所有模块**

删除旧 main() 的内容，替换为：

```python
def main():
    """Morn 进程入口。初始化所有子系统并启动事件循环。"""
    args = parse_args()
    instance_name = args.instance
    data_dir = Path.home() / ".morn" / "instances" / instance_name
    data_dir.mkdir(parents=True, exist_ok=True)
    
    # 1. 初始化状态
    state = MornState(
        instance_name=instance_name,
        data_dir=data_dir,
    )
    
    # 2. 配置
    config = load_config(data_dir)
    state.config = config
    state.log("server", f"config loaded: mode={config['mode']}")
    
    # 3. 初始化日志
    logger = setup_logging(data_dir)
    state.log("server", f"starting instance [{instance_name}]...")
    
    # 4. 初始化 MemoryStore
    async def init_subsystems(state):
        try:
            store = MemoryStore(data_dir)
            await store.__aenter__()
            state.memory_store = store
            state.db = store.db  # 让心跳循环可以访问
            state.memory_count = await store.count()
            state.log("memory", f"loaded {state.memory_count} capsules")
        except Exception as e:
            state.log("memory", f"init failed: {e}")
        
        # 5. 初始化 ChatEngine
        try:
            if state.memory_store:
                state.chat_engine = ChatEngine(
                    instance_name=instance_name,
                    memory_store=state.memory_store,
                    config=config,
                )
                state.log("chat", "chat engine initialized")
        except Exception as e:
            state.log("chat", f"init failed: {e}")
        
        # 6. 初始化 UserProtection
        try:
            state.protection = UserProtection()
            state.log("security", "user protection loaded")
        except Exception as e:
            state.log("security", f"init failed: {e}")
    
    # 7. 异步启动
    async def async_main():
        await init_subsystems(state)
        
        tasks = [
            asyncio.create_task(heartbeat_loop(state), name="morn-heartbeat"),
            asyncio.create_task(memory_monitor(state), name="morn-memmon"),
            asyncio.create_task(wal_checkpoint(state), name="morn-wal"),
            asyncio.create_task(cli_loop(state), name="morn-cli"),
        ]
        
        # 8. Telegram Bot（如果配置了 token）
        if config.get("telegram_token") and _HAS_TELEGRAM:
            try:
                state.telegram_bot = TelegramBot(
                    token=config["telegram_token"],
                    chat_engine=state.chat_engine,
                    memory_store=state.memory_store,
                    data_dir=data_dir,
                    instance_name=instance_name,
                )
                tasks.append(
                    asyncio.create_task(state.telegram_bot.start(), name="morn-telegram")
                )
                state.log("telegram", "bot started")
            except Exception as e:
                state.log("telegram", f"init failed: {e}")
        
        # 9. 打印启动信息
        print(f"\nMorn v0.1 [{instance_name}] 已启动")
        print(f"  PID: {os.getpid()}")
        print(f"  数据目录: {data_dir}")
        print(f"  记忆: {state.memory_count if state.memory_store else 0} 条")
        if state.chat_engine:
            print(f"  情感: {state.chat_engine.emotion}")
        print(f"  Telegram: {'已连接' if state.telegram_bot else '未配置（仅命令行模式）'}")
        print(f"  模型: {config.get('mode', 'hybrid')}")
        print(f"\n输入 /status 查看状态，/shutdown 退出\n")
        
        await asyncio.gather(*tasks, return_exceptions=True)
        
        # 10. 退出清理
        if state.telegram_bot:
            await state.telegram_bot.stop()
        if state.memory_store:
            await state.memory_store.close()
        
        if state.heartbeat_count > 0:
            print(f"[Morn] 本次运行: {state.heartbeat_count} 次心跳 | 再见。")
    
    # 11. 注册信号处理器
    def handle_signal(signum, frame):
        print("\n[Morn] 正在关闭...")
        state.shutdown = True
    
    signal.signal(signal.SIGINT, handle_signal)
    signal.signal(signal.SIGTERM, handle_signal)
    
    # 12. 启动事件循环
    asyncio.run(async_main())
```

"""集成测试：验证所有模块可串联。"""

import json
import os
import sys
import tempfile
from pathlib import Path

import pytest

sys.path.insert(0, os.path.join(os.path.dirname(__file__), ".."))


class TestConfig:
    """配置加载测试"""

    def test_load_config_creates_default(self):
        """config.json 不存在时创建默认配置"""
        from morn_core.server import load_config
        
        with tempfile.TemporaryDirectory() as tmpdir:
            data_dir = Path(tmpdir)
            config = load_config(data_dir)
            
            assert config["instance_name"] == "default"
            assert config["instance_type"] == "平衡型"
            assert config["mode"] == "hybrid"
            assert config["memory_retention_days"] == 30
            assert config["birth_completed"] is False
            
            # 确认文件已创建
            config_path = data_dir / "config.json"
            assert config_path.exists()

    def test_load_config_loads_existing(self):
        """config.json 存在时加载已有配置"""
        from morn_core.server import load_config
        
        with tempfile.TemporaryDirectory() as tmpdir:
            data_dir = Path(tmpdir)
            custom = {"instance_name": "test_morn", "mode": "local"}
            config_path = data_dir / "config.json"
            with open(config_path, "w") as f:
                json.dump(custom, f)
            
            config = load_config(data_dir)
            assert config["instance_name"] == "test_morn"
            assert config["mode"] == "local"

    def test_load_config_fills_missing_defaults(self):
        """config.json 缺失字段用默认值补充"""
        from morn_core.server import load_config
        
        with tempfile.TemporaryDirectory() as tmpdir:
            data_dir = Path(tmpdir)
            partial = {"instance_name": "partial"}
            config_path = data_dir / "config.json"
            with open(config_path, "w") as f:
                json.dump(partial, f)
            
            config = load_config(data_dir)
            assert config["instance_name"] == "partial"
            # 缺失字段被默认值补充
            assert config["mode"] == "hybrid"
            assert config["memory_retention_days"] == 30

class TestEndToEnd:
    """端到端基本流程测试（不启动事件循环）"""

    @pytest.mark.asyncio
    async def test_memory_chat_protection_flow(self):
        """记忆 → 对话 → 保护层，数据流可跑通"""
        from morn_core.memory.store import MemoryStore
        from morn_core.chat.engine import ChatEngine, EmotionState
        from morn_core.security.user_protection import UserProtection
        
        with tempfile.TemporaryDirectory() as tmpdir:
            data_dir = Path(tmpdir)
            
            # 1. 初始化 MemoryStore
            async with MemoryStore(data_dir) as store:
                # 2. 初始化 ChatEngine
                engine = ChatEngine(
                    instance_name="test",
                    memory_store=store,
                    config={"mode": "local"},
                )
                
                # 3. 写入一条记忆
                eid = await store.add_capsule({
                    "entities": '["创建者", "Morn"]',
                    "description": "创建者说正在测试系统",
                })
                assert eid is not None
                
                # 4. 检索记忆
                results = await store.search_fts("测试")
                assert len(results) >= 1
                
                # 5. 验证情感状态机可工作
                engine.emotion.apply_delta(0.3, "高兴/满足")
                assert engine.emotion.pleasure > 0.5
                
                # 6. 验证保护层可工作
                protection = UserProtection()
                filtered, triggered = protection.filter("你好，今天天气不错")
                assert triggered == []
                
                # 7. 验证计数
                count = await store.count()
                assert count >= 1
                
                print("端到端数据流验证通过 ✅")

    def test_command_line_args(self):
        """命令行参数解析"""
        from morn_core.server import parse_args
        
        old_argv = sys.argv
        try:
            sys.argv = ["server.py", "--instance", "test_e2e"]
            args = parse_args()
            assert args.instance == "test_e2e"
        finally:
            sys.argv = old_argv
