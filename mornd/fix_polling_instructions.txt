# aiogram polling 修复

## 问题
aiogram 3.x 的 `dp.start_polling()` 在 WSL 代理环境下启动后立即停止（日志：Start polling → Polling stopped），无法接收消息。

## 修复方案
移除 aiogram 的 Dispatcher 和 start_polling，改为手动 getUpdates 长轮询。

## 修改文件
mornd/morn_core/presence/telegram_bot.py

## 具体修改

### 1. 修改导入
保留：
- import json, logging, os, asyncio
- from pathlib import Path

- from aiogram.types import Message（仅用于类型提示）
- from aiogram.client.default import DefaultBotProperties
- from aiohttp import ClientSession

移除：
- from aiogram import Dispatcher
- from aiogram.filters import Command

新增：
- import asyncio

### 2. 修改 __init__ 方法

当前：
```python

self.dp = Dispatcher()
self.birth_guide = BirthGuide(config_path)
```

改为：

```python
proxy = os.environ.get("HTTPS_PROXY") or os.environ.get("HTTP_PROXY")
session_kwargs = {}

    session_kwargs["proxy"] = proxy
```

### 3. 删除 _register_handlers 方法

不再使用 aiogram 的 handler 注册方式。所有命令处理逻辑移到 _dispatch 方法中。

### 4. 新增 _dispatch 方法

```python
async def _dispatch(self, msg_data: dict):
    """分发 Telegram 消息到对应的处理函数"""
    text = msg_data.get("text", "")
    chat_id = msg_data["chat"]["id"]
    message_id = msg_data.get("message_id", 0)
    
    # 构造一个简单的 message 对象（用 dict 替代 aiogram Message）
    # 直接用 chat_id 操作
    
    self._logger.info(f"dispatch: chat={chat_id} text={text[:50]}")
    
    # 命令分发
    if text.startswith("/"):
        parts = text.split()
        cmd = parts[0].lower()
        
        if cmd == "/start":
            await self._cmd_start(chat_id)
        elif cmd == "/status":
            await self._cmd_status(chat_id)
        elif cmd == "/forget":
            await self._cmd_forget(chat_id, parts[1] if len(parts) > 1 else "")
        elif cmd == "/clear":
            await self._cmd_clear(chat_id)
        elif cmd == "/help":
            await self._cmd_help(chat_id)
        # 未知命令走普通消息
        else:
            await self._handle_message(chat_id, text)
    else:
        await self._handle_message(chat_id, text)
```

```python
async def _cmd_clear(self, chat_id: int):
    if not hasattr(self, "_pending_clear") or self._pending_clear != chat_id:
        self._pending_clear = chat_id
        await self.bot.send_message(chat_id,
            "⚠️ 这将删除你和我之间的所有对话记忆，不可恢复。\n\n"
            "再次发送 /clear 确认清除，或发送其他消息取消。"

            await self.memory_store.close()
            import os as _os
            _os.remove(str(self.memory_store.db_path))

            await self.bot.send_message(chat_id, "清理时出错，请稍后重试。")
            return
    self.birth_guide.state = "AWAITING_NAME"
    self.birth_guide.data = {"creator_name": "", "instance_type": ""}
    self.birth_guide._save()
    await self.bot.send_message(chat_id, "已清除所有记忆数据。如果你想重新开始，请发送 /start。")
```

_handle_message（普通消息）:
```python
async def _handle_message(self, chat_id: int, text: str):
    # 1. 诞生引导
    if not self.birth_guide.is_completed():
        reply, _ = await self.birth_guide.process(text)
        if reply:
            await self.bot.send_message(chat_id, reply)
        return
    # 2. 正常聊天
    if not self.chat_engine:
        await self.bot.send_message(chat_id, "聊天引擎尚未就绪，请稍后再试。")
        return
    try:
        reply = await self.chat_engine.chat(text)
        if self.chat_engine.protection or True:
            from morn_core.security.user_protection import UserProtection
            # protection 在 server.py 的 state.protection 中，这里用独立实例
            # 实际上 protection 由 server.py 在 cli_loop 中调用，bot 中调用 chat_engine.chat() 后不需要额外保护
        if len(reply) > 4096:
            for i in range(0, len(reply), 4096):
                await self.bot.send_message(chat_id, reply[i:i+4096])
        else:
            await self.bot.send_message(chat_id, reply)
    except Exception as e:
        self._logger.error(f"chat error: {e}")
        await self.bot.send_message(chat_id, "我好像没听清楚，能再说一遍吗？")
```

### 6. 修改 start 方法

```python
async def start(self):
    self._logger.info("Telegram Bot polling started (manual mode)")
    while not self._stop_polling:
        try:
            url = f"https://api.telegram.org/bot{self.token}/getUpdates"
            params = {"offset": self._polling_offset, "timeout": 30}
            async with self.http_session.get(url, params=params, timeout=35) as resp:
                data = await resp.json()
                for update in data.get("result", []):
                    if "message" in update:
                        await self._dispatch(update["message"])
                    self._polling_offset = update["update_id"] + 1
        except asyncio.CancelledError:
            break
        except Exception as e:
            self._logger.error(f"poll error: {e}")
            await asyncio.sleep(2)
    self._logger.info("Telegram Bot polling stopped")
```

### 7. 修改 stop 方法

```python
async def stop(self):
    self._logger.info("Telegram Bot stopping...")
    self._stop_polling = True
    await self.http_session.close()
    await self.bot.session.close()
```
