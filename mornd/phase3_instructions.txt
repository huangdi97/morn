### MemoryStore 类

```python
class MemoryStore:
    """
    Morn 记忆存储层。
    管理事件胶囊的 CRUD、FTS5 全文检索、软遗忘机制。
    
    使用方式：
        async with MemoryStore(data_dir) as store:
            event_id = await store.add_capsule({...})
            results = await store.search_fts("关键词")
    
    SQLite 文件路径：{data_dir}/memory.db
    """
    
    def __init__(self, data_dir: Path):
        self.data_dir = Path(data_dir)
        self.db_path = self.data_dir / "memory.db"
        self.db: Optional[aiosqlite.Connection] = None

    async def __aenter__(self) -> "MemoryStore":
        """异步上下文管理器入口：打开数据库连接并初始化 schema"""
        self.data_dir.mkdir(parents=True, exist_ok=True)
        self.db = await aiosqlite.connect(str(self.db_path))
        self.db.row_factory = aiosqlite.Row  # 行按列名访问
        await self.db.execute("PRAGMA journal_mode=WAL")
        await self.db.execute("PRAGMA foreign_keys=ON")
        await self._create_schema()
        return self

    async def __aexit__(self, exc_type, exc_val, exc_tb):
        """异步上下文管理器出口：关闭数据库连接"""
        if self.db:
            await self.db.close()
            self.db = None
```

### _create_schema()

```python
    async def _create_schema(self):
        """创建或迁移数据库 schema（幂等，多次执行不报错）。
        
        执行顺序：
        1. CREATE TABLE IF NOT EXISTS capsules
        2. CREATE VIRTUAL TABLE IF NOT EXISTS capsules_fts
        3. CREATE TRIGGER IF NOT EXISTS (INSERT + DELETE 各一个)
        4. CREATE INDEX IF NOT EXISTS（3 个）
        """
```

#### capsules 表定义

```sql
CREATE TABLE IF NOT EXISTS capsules (
    event_id        TEXT PRIMARY KEY,
    timestamp       TEXT NOT NULL,           -- ISO8601: "2026-05-27T17:30:12Z"
    entities        TEXT NOT NULL DEFAULT '[]',   -- JSON 数组: '["创建者", "Morn"]'
    emotion_score   REAL NOT NULL DEFAULT 0.0,    -- -1.0 ~ 1.0
    emotion_tag     TEXT NOT NULL DEFAULT '',      -- "高兴/满足"
    description     TEXT NOT NULL,                 -- 事件描述文本
    importance_weight REAL NOT NULL DEFAULT 0.5,   -- 0.0 ~ 1.0
    causal_edges    TEXT NOT NULL DEFAULT '[]',    -- JSON 数组: '["evt_xxx", "evt_yyy"]'
    source          TEXT NOT NULL DEFAULT 'chat',  -- 'chat' / 'system' / 'evolution'
    forgotten       INTEGER NOT NULL DEFAULT 0,    -- 0=正常, 1=系统遗忘
    forget_creator  INTEGER NOT NULL DEFAULT 0,    -- 0=正常, 1=创建者请求遗忘
    created_at      TEXT NOT NULL DEFAULT (datetime('now'))
);
```

#### 触发器定义

```sql
-- INSERT 触发器：自动同步到 FTS5
CREATE TRIGGER IF NOT EXISTS capsules_ai AFTER INSERT ON capsules
BEGIN
    INSERT INTO capsules_fts(rowid, event_id, description, entities)
    VALUES (new.rowid, new.event_id, new.description, new.entities);
END;

-- DELETE 触发器：自动从 FTS5 删除
CREATE TRIGGER IF NOT EXISTS capsules_ad AFTER DELETE ON capsules
BEGIN
    INSERT INTO capsules_fts(capsules_fts, rowid, event_id, description, entities)
    VALUES ('delete', old.rowid, old.event_id, old.description, old.entities);
END;
```

### _make_event_id()

```python
    @staticmethod
    def _make_event_id() -> str:
        """生成唯一事件胶囊 ID。
        
        格式：evt_YYYYMMDD_HHMMSS_XXX
        其中 XXX = random.randint(100, 999)
        
        示例：evt_20260527_173012_742
        
        注意：此方法不保证全局唯一，冲突时调用方需重试。
        """
        now = datetime.now(timezone.utc)
        timestamp = now.strftime("%Y%m%d_%H%M%S")
        suffix = random.randint(100, 999)
        return f"evt_{timestamp}_{suffix}"
```

### add_capsule()

```python
    async def add_capsule(self, capsule: dict) -> str:
        """写入一个新事件胶囊。
        
        参数：
            capsule: dict，支持字段见下方默认值。必需字段：description
        
        返回：
            str: 生成的 event_id
        
        异常：
            RuntimeError: 3 次重试后仍失败
        
        默认值填充逻辑：
            event_id: 未提供则 self._make_event_id()
            timestamp: 未提供则 datetime.now(timezone.utc).strftime("%Y-%m-%dT%H:%M:%SZ")
            entities: 未提供则 "[]"
            emotion_score: 未提供则 0.0
            emotion_tag: 未提供则 ""
            description: 必需，缺失则抛 ValueError("description is required")
            importance_weight: 未提供则 0.5
            causal_edges: 未提供则 "[]"
            source: 未提供则 "chat"
            forgotten: 未提供则 0
            forget_creator: 未提供则 0
        
        重试逻辑：
            1. 外层 for attempt in range(3) 循环
            2. 捕获所有 Exception（不只是 IntegrityError）
            3. attempt < 2 时：记录 warning 日志，等待 0.5 秒后重试
            4. attempt == 2（最后一次）：记录 error 日志，
               抛 RuntimeError("add_capsule failed after 3 retries: {原始错误}")
           日志通过 logging.getLogger("morn.memory") 输出
        """
        # 1. 验证 description
        if "description" not in capsule or not capsule["description"].strip():
            raise ValueError("description is required")
        
        # 2. 填充默认值
        event_id = capsule.get("event_id", self._make_event_id())
        timestamp = capsule.get("timestamp",
            datetime.now(timezone.utc).strftime("%Y-%m-%dT%H:%M:%SZ"))
        
        # 3. 构建完整数据
        data = {
            "event_id": event_id,
            "timestamp": timestamp,
            "entities": capsule.get("entities", "[]"),
            "emotion_score": capsule.get("emotion_score", 0.0),
            "emotion_tag": capsule.get("emotion_tag", ""),
            "description": capsule["description"],
            "importance_weight": capsule.get("importance_weight", 0.5),
            "causal_edges": capsule.get("causal_edges", "[]"),
            "source": capsule.get("source", "chat"),
            "forgotten": capsule.get("forgotten", 0),
            "forget_creator": capsule.get("forget_creator", 0),
        }
        
        # 4. 执行 INSERT，最多重试 3 次
        for attempt in range(3):
            try:
                await self.db.execute("""
                    INSERT INTO capsules
                    (event_id, timestamp, entities, emotion_score, emotion_tag,
                     description, importance_weight, causal_edges, source,
                     forgotten, forget_creator)
                    VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
                """, (
                    data["event_id"], data["timestamp"], data["entities"],
                    data["emotion_score"], data["emotion_tag"],
                    data["description"], data["importance_weight"],
                    data["causal_edges"], data["source"],
                    data["forgotten"], data["forget_creator"],
                ))
                await self.db.commit()
                return data["event_id"]
            except Exception as e:
                if attempt < 2:
                    logger = logging.getLogger("morn.memory")
                    logger.warning("add_capsule attempt %d failed: %s", attempt + 1, e)
                    # 生成新的 event_id 防止主键冲突
                    data["event_id"] = self._make_event_id()
                    await asyncio.sleep(0.5)
                else:
                    logger = logging.getLogger("morn.memory")
                    logger.error("add_capsule failed after 3 attempts: %s", e)
                    raise RuntimeError(f"add_capsule failed after 3 retries: {e}")
```

### search_fts()

```python
    async def search_fts(self, query: str, limit: int = 10) -> list[dict]:
        """FTS5 全文检索。
        
        参数：
            query: str，FTS5 查询语法（支持 AND/OR 等，见 SQLite FTS5 文档）
                  注意：FTS5 默认用双引号括短语，如 '"衰老干预"'
            limit: int，最多返回条数，默认 10
        
        返回：
            list[dict]，按 FTS5 rank 排序
        
        实现：
        1. SQL 用 JOIN capsules_fts：
           SELECT c.* FROM capsules c
           JOIN capsules_fts f ON c.event_id = f.event_id
           WHERE capsules_fts MATCH ? AND c.forget_creator = 0
           ORDER BY rank
           LIMIT ?
        
        2. 注意：MATCH 语法要求 query 中不含非法字符（如未闭合的引号）
           如果 MATCH 抛 aiosqlite.OperationalError，捕获后返回空列表
           （不要在 retrieve 时让用户看到数据库错误）
        """
        try:
            cursor = await self.db.execute("""
                SELECT c.* FROM capsules c
                JOIN capsules_fts f ON c.event_id = f.event_id
                WHERE capsules_fts MATCH ? AND c.forget_creator = 0
                ORDER BY rank
                LIMIT ?
            """, (query, limit))
            rows = await cursor.fetchall()
            return [dict(row) for row in rows]
        except aiosqlite.OperationalError:
            # FTS5 查询语法错误（如未闭合的引号），安全返回空列表
            return []
```

```python
    async def cleanup_expired(self, retention_days: int = 30):
```

```python
@pytest.mark.asyncio
async def test_cleanup_expired(data_dir):
    """过期清理标记（不删除 chat 源 30 天前的记忆，但保留 evolution 源）"""
    async with MemoryStore(data_dir) as store:
        # 直接 SQL 插入一条过去 31 天的记录
        import datetime
        old_time = (datetime.datetime.now(datetime.timezone.utc) -
                    datetime.timedelta(days=31))
        old_ts = old_time.strftime("%Y-%m-%dT%H:%M:%SZ")
        await store.db.execute("""
            INSERT INTO capsules (event_id, timestamp, description, source)
            VALUES (?, ?, ?, 'chat')
        """, ("evt_old", old_ts, "old memory"))
        await store.db.commit()
        
        # 再插入一条 evolution 源的老记录
        await store.db.execute("""
            INSERT INTO capsules (event_id, timestamp, description, source)
            VALUES (?, ?, ?, 'evolution')
        """, ("evt_old_evo", old_ts, "old evolution"))
        await store.db.commit()
        
        await store.cleanup_expired(retention_days=30)
        
        # chat 源应被标记
        cap = await store.get_capsule("evt_old")
        assert cap["forgotten"] == 1
        
        # evolution 源应保留
        cap_evo = await store.get_capsule("evt_old_evo")
        assert cap_evo["forgotten"] == 0
```
