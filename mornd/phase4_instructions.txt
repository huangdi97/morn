class EmotionState:
    """
    三维情感状态机。
    
    维护三个维度：
    - calmness（平静度）：0~1，默认 0.7，对话基调
    - pleasure（愉悦度）：0~1，默认 0.5，对话情绪
    - connection（联结感）：0~1，默认 0.3，亲密程度
    
    每个维度有基线值（默认值）和当前值。
    所有维度值自动钳制在 [0, 1] 区间。
    """
    
    def __init__(self):
        self.calmness: float = 0.7
        self.pleasure: float = 0.5
        self.connection: float = 0.3
        # 基线值（衰减目标）
        self._calmness_baseline: float = 0.7
        self._pleasure_baseline: float = 0.5
        self._connection_baseline: float = 0.3
    
    def apply_delta(self, delta_score: float, tag: str = ""):
        """
        根据情感 delta 更新状态。
        LLM 输出 emotion_tag 时调用此方法。
        
        规则：
        - delta_score > 0（正面）：pleasure += delta_score * 0.3
                                 calmness += delta_score * 0.1
                                 connection += delta_score * 0.2
        - delta_score < 0（负面）：pleasure += delta_score * 0.5
                                 calmness += delta_score * 0.15
                                 connection += delta_score * 0.1
        - tag == "惊喜/感动"：额外 +0.1 到 pleasure 和 connection
        - tag 含 "失望/沮丧"：额外 -0.05 到 pleasure
        - 所有值 clamp 到 [0, 1]
        
        参数：
        delta_score: float, -1.0 ~ 1.0
        tag: str, 情感标签文本（可选，用于特殊修正）
        """
        if delta_score > 0:
            self.pleasure += delta_score * 0.3
            self.calmness += delta_score * 0.1
            self.connection += delta_score * 0.2
        else:
            self.pleasure += delta_score * 0.5
            self.calmness += delta_score * 0.15
            self.connection += delta_score * 0.1
        
        # 特殊标签修正
        if tag in ("惊喜/感动", "惊喜", "感动"):
            self.pleasure = min(1.0, self.pleasure + 0.1)
            self.connection = min(1.0, self.connection + 0.1)
        if "失望" in tag or "沮丧" in tag:
            self.pleasure = max(0.0, self.pleasure - 0.05)
        
        self._clamp()
    
    def decay(self):
        """
        自然衰减。
        所有维度趋近基线值，每 5 分钟调用一次。
        
        规则：
        calmness += (self._calmness_baseline - self.calmness) * 0.1
        pleasure += (self._pleasure_baseline - self.pleasure) * 0.05
        connection += (self._connection_baseline - self.connection) * 0.02
        clamp 到 [0, 1]
        """
        self.calmness += (self._calmness_baseline - self.calmness) * 0.1
        self.pleasure += (self._pleasure_baseline - self.pleasure) * 0.05
        self.connection += (self._connection_baseline - self.connection) * 0.02
        self._clamp()
    
    def _clamp(self):
        """将所有维度钳制到 [0, 1]"""
        self.calmness = max(0.0, min(1.0, self.calmness))
        self.pleasure = max(0.0, min(1.0, self.pleasure))
        self.connection = max(0.0, min(1.0, self.connection))
    
    def __repr__(self) -> str:
        return f"calmness={self.calmness:.1f}, pleasure={self.pleasure:.1f}, connection={self.connection:.1f}"
    
    def to_dict(self) -> dict:
        return {
            "calmness": round(self.calmness, 2),
            "pleasure": round(self.pleasure, 2),
            "connection": round(self.connection, 2),
        }
```

class ChatEngine:
    """
    Morn 对话引擎。
    
    职责：
    - 记忆增强对话（检索相关记忆并入 LLM context）
    - 云端/本地模型路由（hybrid 模式自动切换）
    - 情感状态维护（调用 EmotionState）
    - 情感标签生成（LLM 输出后标注 delta）
    - 对话历史写入事件胶囊
    
    使用方式：
        engine = ChatEngine(instance_name="test", memory_store=store, config={...})
        reply = await engine.chat("你好")
    """
    
    def __init__(self, instance_name: str, memory_store, config: dict):
        """
        参数：
        instance_name: str, 实例名称（用于 prompt）
        memory_store: MemoryStore 实例（外部传入）
        config: dict, 配置项（见 load_config 的默认值）
        """
        self.instance_name = instance_name
        self.memory_store = memory_store
        self.config = config
        self.emotion = EmotionState()
        self._instance_type = config.get("instance_type", "平衡型")
        self._mode = config.get("mode", "hybrid")
        self._logger = logging.getLogger("morn.chat")
    
    async def chat(self, user_message: str) -> str:
        """
        处理用户消息，返回回复。
        
        流程：
        1. 从 memory_store 检索相关记忆
        2. 组装 system prompt + 记忆上下文 + 用户消息
        3. 调用 LLM（根据 mode 选择云端/本地）
        4. 生成情感标签（LLM 输出后标注）
        5. 将交互（用户消息+回复+情感）写入事件胶囊
        6. 返回回复文本
        
        异常处理：
        - LLM 调用失败（网络超时/API 错误）：返回降级回复
        - memory_store 检索失败：空上下文继续对话
        """
        # 1. 检索记忆
        memories = await self._search_memory(user_message)
        
        # 2. 组装 prompt
        messages = await self._assemble_prompt(user_message, memories)
        
        # 3. 调用 LLM
        try:
            reply = await self._call_llm(messages)
        except Exception as e:
            self._logger.error(f"LLM call failed: {e}")
            reply = "我现在有点连接不上。你可以等一下再试，或者换个话题继续。" \
                    if self._mode == "hybrid" else \
                    "我好像没听清楚，能再说一遍吗？"
        
        # 4. 情感标签生成
        try:
            delta, tag = await self._generate_emotion_tag(user_message, reply)
            self.emotion.apply_delta(delta, tag)
        except Exception as e:
            self._logger.warning(f"emotion tag generation failed: {e}")
            delta, tag = 0.0, ""
        
        # 5. 写入记忆
        try:
            await self.memory_store.add_capsule({
                "entities": json.dumps(["创建者", self.instance_name]),
                "emotion_score": self.emotion.pleasure,
                "emotion_tag": tag,
                "description": f"创建者: {user_message[:100]} | Morn: {reply[:100]}",
                "source": "chat",
            })
        except Exception as e:
            self._logger.error(f"memory write failed: {e}")
        
        return reply
```

### _search_memory()

```python
    async def _search_memory(self, query: str) -> str:
        """
        从 memory_store 搜索相关记忆，返回格式化的上下文文本。
        
        搜索策略：
        1. 如果 query 长度 > 3，用 FTS5 搜索（取最近 5 条匹配）
        2. 同时取最近 3 条记忆作为补充上下文
        3. 合并去重（按 event_id 去重）
        4. 格式化：
           相关记忆：
           [3天前] 创建者提到他正在研究衰老干预。
           [1小时前] 创建者说今天心情很好。
        
        如果检索失败（store 为 None 或异常），返回空字符串。
        """
        if self.memory_store is None:
            return ""
        
        try:
            memories = []
            seen_ids = set()
            
            # FTS 搜索
            if len(query) > 3:
                fts_results = await self.memory_store.search_fts(query, limit=5)
                for cap in fts_results:
                    if cap["event_id"] not in seen_ids:
                        memories.append(cap)
                        seen_ids.add(cap["event_id"])
            
            # 补充最近记忆
            recent = await self.memory_store.get_recent(limit=3)
            for cap in recent:
                if cap["event_id"] not in seen_ids:
                    memories.append(cap)
                    seen_ids.add(cap["event_id"])
            
            if not memories:
                return ""
            
            # 格式化
            lines = ["相关记忆："]
            for cap in memories:
                ts = cap.get("timestamp", "")[:10]  # YYYY-MM-DD
                desc = cap.get("description", "")[:80]
                lines.append(f"[{ts}] {desc}")
            
            return "\n".join(lines)
        except Exception:
            return ""
```

回复要求：
- 简洁自然，有温度但不煽情
- 中文回答
- 50 字以内（除非创建者提出了复杂问题）

### _call_llm()

```python
    async def _call_llm(self, messages: list) -> str:
        """
        调用 LLM 获取回复。根据 config 中的 mode 选择路由。
        
        路由逻辑：
        1. mode == "local"：走 _call_local()
        2. mode == "cloud"：走 _call_cloud()，失败则抛异常（不降级）
        3. mode == "hybrid"（默认）：
           a. 先检查网络连通性（HEAD https://api.deepseek.com/ping）
           b. 网络可用 → _call_cloud()
           c. 网络不可用 → _call_local()
           d. 云端失败 → 降级到 _call_local()
        
        异常：所有 LLM 调用失败时抛异常，由 chat() 上层捕获处理
        """
        if self._mode == "local":
            return await self._call_local(messages)
        
        if self._mode == "cloud":
            return await self._call_cloud(messages)
        
        # hybrid 模式
        if await self._check_network():
            try:
                return await self._call_cloud(messages)
            except Exception:
                self._logger.info("cloud failed, falling back to local")
                return await self._call_local(messages)
        else:
            self._logger.info("network unavailable, using local model")
            return await self._call_local(messages)
```

### _check_network()

```python
    async def _check_network(self, timeout: int = 3) -> bool:
        """
        检查网络连通性。
        
        实现：
        1. 用 aiohttp 发 HEAD 请求到 https://api.deepseek.com/ping
        2. timeout=3 秒
        3. 状态码 >= 200 且 < 500 视为可用
        4. 超时或连接错误返回 False
        """
        try:
            async with aiohttp.ClientSession() as session:
                async with session.head(
                    "https://api.deepseek.com/ping",
                    timeout=aiohttp.ClientTimeout(total=timeout)
                ) as resp:
                    return 200 <= resp.status < 500
        except Exception:
            return False
```

        5. 解析 response.json()["choices"][0]["message"]["content"]
        6. 非 200 状态码抛 RuntimeError(f"API error: {status} {text}")
        7. 网络异常原样上抛
        """

    async def _call_local(self, messages: list) -> str:
        """
        调用本地 Ollama 模型。

### _generate_emotion_tag()

```python
    async def _generate_emotion_tag(self, user_msg: str, reply: str) -> tuple:
        """
        基于对话内容生成情感标签（delta 模式）。
        
        delta 模式说明：
        - 状态机维护独立状态值
        - LLM 只输出变化量（delta），不输出绝对值
        - 一致性由状态机保证，LLM 只提供方向
        
        实现：
        1. 组装一个小 prompt 给 LLM：
           分析以下对话，输出情感变化量（delta_score, -1~1）和标签。
           ===
           用户：{user_msg[:200]}
           回复：{reply[:200]}
           ===
           请只输出 JSON：{{"delta_score": 0.0, "tag": "标签名"}}
           标签从以下选择：高兴/满足 / 惊喜/感动 / 平静 / 好奇 / 担忧 / 失望/沮丧 / 温暖 / 联结
        
        2. 调用 _call_cloud()（用少量 token 的云端调用）
           如果不可用，调用 _call_local()
           如果都不可用，返回 (0.0, "")
        
        3. 解析 JSON，返回 (delta_score, tag)
           解析失败（JSON 格式错误）返回 (0.0, "")
        """
        # 构建情感分析 prompt
        emotion_prompt = [
            {"role": "system", "content": "你是一个情感分析器。只输出 JSON。"},
            {"role": "user", "content": f"分析以下对话的情感变化量：\n用户：{user_msg[:200]}\n回复：{reply[:200]}\n请只输出JSON：{{\"delta_score\": 数值(-1~1), \"tag\": \"标签\"}}"},
        ]
        
        try:
            if await self._check_network():
                text = await self._call_cloud(emotion_prompt)
            else:
                text = await self._call_local(emotion_prompt)
            
            # 提取 JSON
            import re
            json_match = re.search(r'\{.*\}', text, re.DOTALL)
            if json_match:
                data = json.loads(json_match.group())
                return (float(data.get("delta_score", 0)), str(data.get("tag", "")))
        except Exception:
            pass
        
        return (0.0, "")
```

"""对话引擎测试。"""

import os
import sys

sys.path.insert(0, os.path.join(os.path.dirname(__file__), ".."))

from morn_core.chat.engine import EmotionState


class TestEmotionState:
    """情感状态机基本行为测试"""

    def test_initial_values(self):
        e = EmotionState()
        assert e.calmness == 0.7
        assert e.pleasure == 0.5
        assert e.connection == 0.3

    def test_positive_delta_increases_pleasure(self):
        e = EmotionState()
        # 应用正面 delta
        e.apply_delta(0.5, "高兴/满足")
        assert e.pleasure > 0.5
        assert e.calmness >= 0.7  # 正面使 calmness 略微上升
        assert e.connection > 0.3

    def test_negative_delta_decreases_pleasure(self):
        e = EmotionState()
        p_before = e.pleasure
        e.apply_delta(-0.3, "失望/沮丧")
        assert e.pleasure < p_before

    def test_surprise_tag_bonus(self):
        e = EmotionState()
        p_before = e.pleasure
        c_before = e.connection
        e.apply_delta(0.2, "惊喜/感动")
        # 惊喜标签有额外加成
        assert e.pleasure >= p_before + 0.2 * 0.3 + 0.08  # delta * 0.3 + bonus 0.1，留点浮点误差

    def test_values_clamped_to_zero(self):
        e = EmotionState()
        e.apply_delta(-10.0, "极度负面")
        assert e.pleasure >= 0.0
        assert e.calmness >= 0.0
        assert e.connection >= 0.0

    def test_values_clamped_to_one(self):
        e = EmotionState()
        e.apply_delta(10.0, "极度正面")
        assert e.pleasure <= 1.0
        assert e.calmness <= 1.0
        assert e.connection <= 1.0

    def test_decay_towards_baseline(self):
        e = EmotionState()
        # 先把值推到极端
        e.pleasure = 0.9
        e.calmness = 0.2
        e.connection = 0.8
        
        # 衰减一次
        e.decay()
        
        # pleasure 接近 0.5，calmness 接近 0.7，connection 接近 0.3
        assert e.pleasure < 0.9
        assert e.calmness > 0.2
        assert e.connection < 0.8

    def test_repr_contains_values(self):
        e = EmotionState()
        r = repr(e)
        assert "0.7" in r
        assert "0.5" in r
        assert "0.3" in r

    def test_to_dict(self):
        e = EmotionState()
        d = e.to_dict()
        assert "calmness" in d
        assert "pleasure" in d
        assert "connection" in d
        assert isinstance(d["calmness"], float)
